SELECT *,
       (SELECT customer_id
        FROM orders
        WHERE orders.id = order_details.order_id) AS customer_id
FROM order_details;

2.
SELECT * FROM order_details
WHERE order_id IN (
    SELECT id FROM orders
    WHERE shipper_id = 3
);

3.
SELECT order_id, AVG(quantity) AS avg_quantity
FROM (
    SELECT *
    FROM order_details
    WHERE quantity > 10
) AS fil_order_details
GROUP BY order_id;

4.
WITH temp AS (
    SELECT *
    FROM order_details
    WHERE quantity > 10
)
SELECT order_id, AVG(quantity) AS avg_quantity
FROM temp
GROUP BY order_id;

DROP FUNCTION IF EXISTS divide_values;

DELIMITER //

CREATE FUNCTION divide_values(a FLOAT, b FLOAT)
RETURNS FLOAT
DETERMINISTIC
BEGIN
    RETURN a / b;
END //

DELIMITER ;

SELECT quantity, divide_values(quantity, 3) AS result
FROM order_details;
